    # –ü—Ä–æ–≤–µ—Ä–∫–∞ –¥–∞–ª–∏ —Å–º–µ —Å—Ç–∏–≥–Ω–∞–ª–∏ —Ü–µ–ª—Ç–∞

        moves.append(empty_pos - 1)

    # 4. –ü—Ä–æ—Å—Ç–æ –ø—Ä–µ–º–µ—Å—Ç–≤–∞–Ω–µ –Ω–∞ –∂–∞–±–∞ <

        moves.append(empty_pos + 1)

    # –ò–∑–ø—Ä–æ–±–≤–∞–Ω–µ –Ω–∞ –≤—Å–∏—á–∫–∏ –≤—ä–∑–º–æ–∂–Ω–∏ —Ö–æ–¥–æ–≤–µ
    for from_pos in moves:
        new_state = state[:]  # –ö–æ–ø–∏—Ä–∞–Ω–µ –Ω–∞ —Å—ä—Å—Ç–æ—è–Ω–∏–µ—Ç–æ
        new_state[from_pos], new_state[empty_pos] = new_state[empty_pos], new_state[from_pos]
        path.append(''.join(new_state))

        # –†–µ–∫—É—Ä—Å–∏–≤–Ω–æ –∏–∑–≤–∏–∫–≤–∞–Ω–µ
